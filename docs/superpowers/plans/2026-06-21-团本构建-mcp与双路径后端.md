# 构建期 MCP + 双路径后端 + CLI 实现计划（Plan 2/3）

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task。MCP 相关任务（Task 1-3）**额外参考 example-skills:mcp-builder**（schema/annotations/错误信封约定）。Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 把 Plan 1 的读写层 + 检索库包成构建期 MCP 门面 `dicelore_build_*`，并让后端（orchestrator）与裸 Claude Code 两条路都能 **bootstrap 一个构建会话**——与跑团那套物理隔离。

**Architecture:** 镜像运行时 MCP（[mcp/server.ts](../../../packages/core/src/mcp/server.ts) `createMcpServer` + [mcp/main.ts](../../../packages/core/src/mcp/main.ts) stdio 入口）造一套构建版：`BUILD_TOOLS` + `createBuildMcpServer` + `build-main.ts`。orchestrator 加 `BuildHost`（独立类，只挂构建 MCP + 构建 skill prompt，无 rollGate / turn-end / canon-notify），与 `SessionHost` 共享抽出的瘦流式 helper。CLI 加 `build-init` 写构建版 `.claude/`。

**Tech Stack:** TypeScript ESM、`@modelcontextprotocol/sdk`、zod、`@anthropic-ai/claude-agent-sdk`、Hono、vitest。

**上游：** [设计 §5/§6/§7](../specs/2026-06-21-团本构建工具链-mcp与skill-design.md)；依赖 **Plan 1** 的 `src/build/pack` 与 `src/build/retrieval`。

## Global Constraints

- **第一性隔离不变量**：构建会话只见 `BUILD_TOOLS`、跑团会话只见 `TOOLS`，两数组编译期不交叉；`BuildHost` 不挂 `PlayerRollGate` / turn-end / canon-notify。任务含断言此不变量的测试。
- **对齐运行时 MCP §0 约定**（[MCP工具面 §0](../../wiki/04-子系统设计/MCP工具面.md)）：入参 Zod `.strict()`；成功走 `structuredContent`（`outputSchema`）；**错误信封 `{error:{code,message,hint}}` 序列化进 `content[].text`、绝不放 `structuredContent`**（SDK 会拿 structuredContent 校验 outputSchema → 触发 -32602）。
- **复用 `ToolDef` 形状**（[mcp/tooldef.ts](../../../packages/core/src/mcp/tooldef.ts)）：`{ name, title, description, inputSchema, outputSchema, annotations, handler }`；`description` 五段式（功能/Args/Returns/use-don't/错误）。
- **构建 handler 不吃 `DB`**：`ToolDef.handler` 现签名是 `(db, input)`；构建工具改吃 `BuildCtx`（packDir + retrievalDb）。本 plan Task 2 定义独立 `BuildToolDef`（handler 吃 `BuildCtx`），不污染运行时 `ToolDef`。
- **每个 `.ts` 文件头**保留 AGPL 注释块。`import` 用 `.js` 后缀。
- **测试**：core 侧 `cd packages/core && npx vitest run <path>`；orchestrator 侧 `cd apps/orchestrator && npx vitest run <path>`。

```ts
import { z } from "zod";

export const issueSchema = z.object({
  level: z.enum(["error", "warn"]), file: z.string(), msg: z.string(), hint: z.string().optional(),
});
export const writeResultOut = z.object({ file: z.string(), warnings: z.array(issueSchema) });

const vis01 = z.union([z.literal(0), z.literal(1)]);
const vis012 = z.union([z.literal(0), z.literal(1), z.literal(2)]);

export const setManifestIn = z.object({
  id: z.string().optional(), version: z.string().optional(), name: z.string().optional(),
  description: z.string().optional(), flows: z.array(z.string()).optional(),
  clock: z.string().optional(), entry: z.string().optional(),
}).strict();

export const writeWorldIn = z.object({
  path: z.string(), content: z.string(),
  tags: z.array(z.string()).optional(), visible: vis01.optional(),
}).strict();

export const addNpcIn = z.object({
  name: z.string(), persona: z.string(), visible: vis01.optional(),
  sheet: z.array(z.object({ attr: z.string(), value: z.string(), visible: vis012.optional() })).optional(),
}).strict();

export const addPoolIn = z.object({
  pool: z.string(),
  rows: z.array(z.record(z.union([z.string(), z.number()]))).min(1),
}).strict();

export const setParamBandIn = z.object({
  table: z.string(),
  bands: z.array(z.object({
    label: z.string(), min: z.number(), max: z.number(), consequence: z.string(),
  })).min(1),
}).strict();

export const setSheetIn = z.object({
  file: z.string().optional(),
  cells: z.array(z.object({
    entity: z.string(), attr: z.string(), value: z.string(), visible: vis012.optional(),
  })).min(1),
}).strict();

export const writeRuleIn = z.object({
  name: z.string(), content: z.string(), version: z.number().int().min(1),
}).strict();

export const addFrontIn = z.object({
  name: z.string(), clock: z.string(), min: z.number(), max: z.number(),
  mode: z.enum(["once", "repeat"]), visible: vis01.optional(),
  stakes: z.string(), prose: z.string(),
  ladder: z.array(z.object({ at: z.number(), payload: z.string() })),
}).strict();

export const readIn = z.object({
  kind: z.enum(["manifest", "world", "npc", "pool", "param", "sheet", "rule", "front"]),
  id: z.string().optional(),
}).strict();
export const readOut = z.object({
  found: z.boolean(), content: z.string(), meta: z.record(z.string()).optional(),
});

export const validateIn = z.object({}).strict();
export const validateOut = z.object({ ok: z.boolean(), issues: z.array(issueSchema) });
```

```ts
// Copyright (C) 2026 MuLeiSY2021
//
// This file is part of Dicelore.
//
// Dicelore is free software: you can redistribute it and/or modify it under
// the terms of the GNU Affero General Public License as published by the Free
// Software Foundation, either version 3 of the License, or (at your option)
// any later version. See <https://www.gnu.org/licenses/>.

import { McpServer } from "@modelcontextprotocol/sdk/server/mcp.js";
import { BUILD_TOOLS } from "./tools.js";
import type { BuildCtx, BuildToolDef } from "./tooldef.js";

interface Envelope {
  content: { type: "text"; text: string }[];
  structuredContent?: unknown;
  isError?: boolean;
}

// 错误信封:{error} 只进 content[].text、绝不进 structuredContent(SDK 会拿它校验 outputSchema → -32602)。
function errEnvelope(code: string, message: string, hint?: string): Envelope {
  return { content: [{ type: "text", text: JSON.stringify({ error: { code, message, hint } }) }], isError: true };
}

const byName = new Map(BUILD_TOOLS.map((t) => [t.name, t]));

export function invokeBuildTool(ctx: BuildCtx, name: string, args: unknown): Envelope {
  const t: BuildToolDef | undefined = byName.get(name);
  if (!t) return errEnvelope("UNKNOWN_TOOL", `未知构建工具: ${name}`);
  const parsed = t.inputSchema.safeParse(args);
  if (!parsed.success) return errEnvelope("VALIDATION", parsed.error.message, "对照工具 Args 修正入参");
  try {
    const out = t.handler(ctx, parsed.data);
    return { content: [{ type: "text", text: JSON.stringify(out) }], structuredContent: out };
  } catch (e) {
    return errEnvelope("BUILD_ERROR", e instanceof Error ? e.message : String(e));
  }
}

export function createBuildMcpServer(ctx: BuildCtx): McpServer {
  const server = new McpServer({ name: "dicelore", version: "0.0.0" });
  for (const t of BUILD_TOOLS) {
    server.registerTool(
      `dicelore_build_${t.name}`,
      {
        title: t.title,
        description: t.description,
        inputSchema: t.inputSchema.shape,
        outputSchema: t.outputSchema.shape,
        annotations: t.annotations,
      },
      (args: unknown) => invokeBuildTool(ctx, t.name, args) as any,
    );
  }
  return server;
}
```

```ts
// Copyright (C) 2026 MuLeiSY2021
//
// This file is part of Dicelore.
//
// Dicelore is free software: you can redistribute it and/or modify it under
// the terms of the GNU Affero General Public License as published by the Free
// Software Foundation, either version 3 of the License, or (at your option)
// any later version. See <https://www.gnu.org/licenses/>.

import { CLIENT_PROTOCOL, type StreamMessage } from "@dicelore/shared";
import type { GmDriver, TurnInput } from "../gm/GmDriver.js";
import type { WsHub } from "./ws.js";

export interface StreamTurnDeps {
  driver: GmDriver;
  hub: WsHub;
  sessionId: string;
  turnId: string;
}

// 驱动 GmDriver 事件流 → 广播 turn_started + 逐条 narration_commit;遇 error 发 error 并返回 errored。
// 不发 turn_ended——回合收尾(turn-end hook / 单纯结束)由调用者按场景决定(跑团跑 turn-end,构建直接结束)。
export async function streamDriverTurn(deps: StreamTurnDeps, input: TurnInput): Promise<{ seq: number; errored: boolean }> {
  const { hub, sessionId, turnId } = deps;
  const send = (m: StreamMessage) => hub.broadcast(sessionId, m);
  send({ protocol: CLIENT_PROTOCOL, type: "turn_started", turnId });
  let seq = 0;
  try {
    for await (const ev of deps.driver.runTurn(input)) {
      if (ev.type === "narration") {
        seq += 1;
        send({ protocol: CLIENT_PROTOCOL, type: "narration_commit", seq, text: ev.text });
      } else if (ev.type === "error") {
        send({ protocol: CLIENT_PROTOCOL, type: "error", code: "gm_error", message: ev.message });
        return { seq, errored: true };
      } else if (ev.type === "turn_end") {
        break;
      }
    }
  } catch (e) {
    send({ protocol: CLIENT_PROTOCOL, type: "error", code: "driver_error", message: e instanceof Error ? e.message : String(e) });
    return { seq, errored: true };
  }
  return { seq, errored: false };
}
```

```ts
import { Hono } from "hono";
import { MessageRequestSchema } from "@dicelore/shared";
import { getOrCreateBuildHost } from "./session/buildRegistry.js";
import type { BuildHost } from "./session/BuildHost.js";
import type { GmDriver } from "./gm/GmDriver.js";

export interface BuildLiveDeps {
  driverFactory: (host: BuildHost) => GmDriver;
  packDirFor: (id: string) => string; // 构建会话 id → 团本目录
}

// 构建会话 server 面：与跑团 /sessions 完全分离的路由前缀 /build-sessions。
export function createBuildApp(deps: BuildLiveDeps): Hono {
  const app = new Hono();
  const hostDeps = (id: string) => ({ packDir: deps.packDirFor(id), driverFactory: deps.driverFactory });

  app.post("/build-sessions/:id/messages", async (c) => {
    const id = c.req.param("id");
    const body = MessageRequestSchema.parse(await c.req.json());
    const host = getOrCreateBuildHost(id, hostDeps(id));
    const { turnId } = await host.handleMessage(body.text);
    return c.json({ turnId }, 202);
  });

  app.get("/build-sessions/:id", (c) => c.json({ sessionId: c.req.param("id"), kind: "build" }));

  return app;
}
```

```ts
// Copyright (C) 2026 MuLeiSY2021
//
// This file is part of Dicelore.
//
// Dicelore is free software: you can redistribute it and/or modify it under
// the terms of the GNU Affero General Public License as published by the Free

// any later version. See <https://www.gnu.org/licenses/>.

import { cpSync, mkdirSync, writeFileSync, appendFileSync, existsSync } from "node:fs";
import { dirname, join } from "node:path";
import { fileURLToPath } from "node:url";
import { buildMcpJson, buildClaudeMdPointer } from "./templates.js";

const ADAPTER_DIR = dirname(fileURLToPath(import.meta.url));
const PKG_ROOT = join(ADAPTER_DIR, "..", "..");
const BUILD_SKILLS_SRC = join(PKG_ROOT, "build-skills"); // 构建 skill 源(Plan 3 落地)

// 构建版 .claude:构建 MCP + 仅 dicelore-build-pack skill;无运行时 hook(choice/rule/L3 皆跑团概念)。
export function runBuildInit(opts: { projectDir: string }): void {
  const { projectDir } = opts;
  const claudeDir = join(projectDir, ".claude");
  mkdirSync(claudeDir, { recursive: true });

  writeFileSync(join(projectDir, ".mcp.json"), JSON.stringify(buildMcpJson({ packDir: projectDir }), null, 2) + "\n");

  // 只拷构建 skill(Plan 3 提供 build-skills/dicelore-build-pack);源不存在则跳过(Plan 2 独立可跑)。
  const skillsDst = join(claudeDir, "skills");
  mkdirSync(skillsDst, { recursive: true });
  const src = join(BUILD_SKILLS_SRC, "dicelore-build-pack");
  if (existsSync(src)) cpSync(src, join(skillsDst, "dicelore-build-pack"), { recursive: true });

  const claudeMd = join(projectDir, "CLAUDE.md");
  const pointer = "\n" + buildClaudeMdPointer();
  if (existsSync(claudeMd)) appendFileSync(claudeMd, pointer);
  else writeFileSync(claudeMd, pointer.trimStart());
}
```
