# 团本包读写层 + 素材检索库 实现计划（Plan 1/3）

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现团本构建工具链的纯逻辑地基——团本包读写层（CRUD over 文件 + 校验器）与素材检索库最小版（切块 + FTS5/jieba）。

**Architecture:** 全部纯 TS 函数模块，不依赖 Agent SDK / MCP / Web，可单测（镜像内层能力库分层哲学）。读写层吃进所有格式细节（CSV 列序、YAML frontmatter、子目录布局），文件即真相、即写即读、无内存态（[ADR-0015](../../wiki/05-决策记录-ADR/README.md)）。检索库复用 core 现有 FTS5/jieba 基建（[store/fts.ts](../../../packages/core/src/store/fts.ts)），切块入独立临时 SQLite。

**Tech Stack:** TypeScript (ESM, `.js` import 后缀)、vitest、better-sqlite3、`@node-rs/jieba`。**零新依赖**——manifest / 元信息全用 markdown 约定（H1 + `键: 值` 列表 + markdown 表格），自己解析，不引 YAML 库。

**上游设计：** [团本构建工具链-mcp与skill 设计 §3/§4](../specs/2026-06-21-团本构建工具链-mcp与skill-design.md)、[组件6 团本与 manifest](../../wiki/04-子系统设计/团本与manifest.md)（产物契约）。

## Global Constraints

- **包目录布局**严格对齐 [组件6 §1](../../wiki/04-子系统设计/团本与manifest.md)：`manifest.md` / `world/**/*.md` / `pools/*.csv` / `params/*.csv` / `rules/*.md` / `sheets/*.csv` / `fronts/*.md`。
- **去 YAML（2026-06-21 格式修订）**：原组件6 的 `manifest.yaml` + YAML frontmatter **改纯 markdown**——`manifest.md`（H1=name、首段=description、`键: 值` 无序列表=标量、`## flows` 列表）；doc 元信息（visible/tags/source/version、front 钟参数）落 **H1 后 `键: 值` 无序列表**（缺省值不写）；front 凶兆阶梯用 **markdown 表格**。**pools/params/sheets 仍 CSV**。

- **编码**：UTF-8；写文件统一 `\n` 行尾、文件末尾留一个换行。

- **每个 `.ts` 文件头**保留现有 AGPL 版权注释块（照抄 [store/world.ts:1-8](../../../packages/core/src/store/world.ts#L1-L8)）。
- **import 用 `.js` 后缀**（ESM NodeNext，照现有代码）。
- **测试命令**：`cd packages/core && npx vitest run <path>`；类型检查 `npx tsc --noEmit`。
- **零新依赖**：不引 YAML 库；markdown 元信息自解析（Task 3）。

```ts
// Copyright (C) 2026 MuLeiSY2021
//
// This file is part of Dicelore.
//
// Dicelore is free software: you can redistribute it and/or modify it under
// the terms of the GNU Affero General Public License as published by the Free
// Software Foundation, either version 3 of the License, or (at your option)
// any later version. See <https://www.gnu.org/licenses/>.

export function csvField(v: string | number): string {
  const s = String(v);
  if (/[",\n]/.test(s)) return `"${s.replace(/"/g, '""')}"`;
  return s;
}

export function serializeCsv(rows: Record<string, string | number>[], columns?: string[]): string {
  let cols = columns;
  if (!cols) {
    const seen = new Set<string>();
    for (const r of rows) for (const k of Object.keys(r)) seen.add(k);
    cols = [...seen];
  }
  const header = cols.map(csvField).join(",");
  const body = rows.map((r) => cols!.map((c) => csvField(r[c] ?? "")).join(",")).join("\n");
  return rows.length ? `${header}\n${body}\n` : `${header}\n`;
}

// 解析支持引号字段(含内部 "" 转义、逗号、换行)。
export function parseCsv(text: string): Record<string, string>[] {
  const rows: string[][] = [];
  let field = "", row: string[] = [], inQ = false;
  for (let i = 0; i < text.length; i++) {
    const c = text[i];
    if (inQ) {
      if (c === '"') {
        if (text[i + 1] === '"') { field += '"'; i++; }
        else inQ = false;
      } else field += c;
    } else if (c === '"') inQ = true;
    else if (c === ",") { row.push(field); field = ""; }
    else if (c === "\n") { row.push(field); rows.push(row); field = ""; row = []; }
    else if (c === "\r") { /* skip */ }
    else field += c;
  }
  if (field !== "" || row.length) { row.push(field); rows.push(row); }
  if (rows.length === 0) return [];
  const header = rows[0];
  return rows.slice(1).filter((r) => r.some((v) => v !== "")).map((r) => {
    const o: Record<string, string> = {};
    header.forEach((h, i) => { o[h] = r[i] ?? ""; });
    return o;
  });
}
```

```ts
// Copyright (C) 2026 MuLeiSY2021
//
// This file is part of Dicelore.
//
// Dicelore is free software: you can redistribute it and/or modify it under
// the terms of the GNU Affero General Public License as published by the Free
// Software Foundation, either version 3 of the License, or (at your option)
// any later version. See <https://www.gnu.org/licenses/>.

export function extractH1(text: string): string | null {
  const m = /^#\s+(.+)$/m.exec(text);
  return m ? m[1].trim() : null;
}

// 首个 `## ` 之前的区域(head):标量元信息列表只在此处取,避免误吃 ## 节下的列表。
function headRegion(text: string): string {
  const idx = text.search(/^##\s/m);
  return idx === -1 ? text : text.slice(0, idx);
}

export function parseKvBullets(text: string): Record<string, string> {
  const out: Record<string, string> = {};
  const re = /^-\s+([A-Za-z_][\w]*)\s*:\s*(.+)$/gm;
  let m: RegExpExecArray | null;
  const head = headRegion(text);
  while ((m = re.exec(head))) out[m[1]] = m[2].trim();
  return out;
}

export function composeKvBullets(meta: Record<string, string | number | undefined>): string {
  return Object.entries(meta)
    .filter(([, v]) => v !== undefined && v !== "")
    .map(([k, v]) => `- ${k}: ${v}`)
    .join("\n");
}

// 定位 `## <heading>` 节的正文行(到下一个 ## 或文末)。
function sectionLines(text: string, heading: string): string[] {
  const lines = text.split("\n");
  const buf: string[] = [];
  let inSec = false;
  for (const line of lines) {
    if (/^##\s+/.test(line)) { inSec = line.replace(/^##\s+/, "").trim() === heading; continue; }
    if (inSec) buf.push(line);
  }
  return buf;
}

export function parseListSection(text: string, heading: string): string[] {
  return sectionLines(text, heading)
    .map((l) => /^-\s+(.+)$/.exec(l.trim()))
    .filter((m): m is RegExpExecArray => m !== null)
    .map((m) => m[1].trim());
}

export function composeListSection(heading: string, items: string[]): string {
  if (items.length === 0) return "";
  return `## ${heading}\n${items.map((i) => `- ${i}`).join("\n")}`;
}

export function parseMdTable(text: string, heading: string): Record<string, string>[] {
  const rows = sectionLines(text, heading).filter((l) => l.trim().startsWith("|"));
  if (rows.length < 2) return [];
  const cols = rows[0].split("|").slice(1, -1).map((s) => s.trim());
  return rows.slice(2).map((r) => {  // rows[1] 是 `| --- | --- |` 分隔行
    const cells = r.split("|").slice(1, -1).map((s) => s.trim());
    const o: Record<string, string> = {};
    cols.forEach((c, i) => { o[c] = cells[i] ?? ""; });
    return o;
  });
}

export function composeMdTable(rows: Record<string, string | number>[], columns: string[]): string {
  const head = `| ${columns.join(" | ")} |`;
  const sep = `| ${columns.map(() => "---").join(" | ")} |`;
  const body = rows.map((r) => `| ${columns.map((c) => String(r[c] ?? "")).join(" | ")} |`).join("\n");
  return rows.length ? `${head}\n${sep}\n${body}` : `${head}\n${sep}`;
}
```

```ts
export interface ValidationIssue {
  level: "error" | "warn";
  file: string;
  msg: string;
  hint?: string;
}

// 所有写函数的统一返回:落的文件 + 该文件局部校验告警(warn 不阻断写)。
export interface WriteResult {
  file: string;
  warnings: ValidationIssue[];
}

export function checkCsvColumns(
  file: string,
  rows: Record<string, string | number>[],
  required: string[],
): ValidationIssue[] {
  const issues: ValidationIssue[] = [];
  if (rows.length === 0) return issues;
  const keys = new Set(Object.keys(rows[0]));
  for (const col of required) {
    if (!keys.has(col)) {
      issues.push({ level: "error", file, msg: `缺必需列「${col}」`, hint: `表头需含 ${required.join(",")}` });
    }
  }
  rows.forEach((r, i) => {
    if (r.weight !== undefined && r.weight !== "" && isNaN(Number(r.weight))) {
      issues.push({ level: "error", file, msg: `第${i + 1}行 weight 非数值: ${r.weight}` });
    }
    if (r.visible !== undefined && r.visible !== "" && ![0, 1, 2].includes(Number(r.visible))) {
      issues.push({ level: "error", file, msg: `第${i + 1}行 visible 非法(须 0/1/2): ${r.visible}` });
    }
  });
  return issues;
}

export function checkRangeBands(
  file: string,
  bands: { label: string; min: number; max: number }[],
): ValidationIssue[] {
  const issues: ValidationIssue[] = [];
  const sorted = [...bands].sort((a, b) => a.min - b.min);
  for (const b of sorted) {
    if (b.min > b.max) issues.push({ level: "error", file, msg: `档「${b.label}」min>max(${b.min}>${b.max})` });
  }
  for (let i = 1; i < sorted.length; i++) {
    const prev = sorted[i - 1], cur = sorted[i];
    if (cur.min <= prev.max) {
      issues.push({ level: "error", file, msg: `档「${prev.label}」与「${cur.label}」区间重叠`, hint: "相邻档 min 应 = 前档 max+1" });
    } else if (cur.min > prev.max + 1) {
      issues.push({ level: "error", file, msg: `档「${prev.label}」与「${cur.label}」间有空洞(未全覆盖)`, hint: "区间须连续无缝" });
    }
  }
  return issues;
}
```

```ts
// Copyright (C) 2026 MuLeiSY2021
//
// This file is part of Dicelore.
//
// Dicelore is free software: you can redistribute it and/or modify it under
// the terms of the GNU Affero General Public License as published by the Free

import { readFileSync, writeFileSync, existsSync, mkdirSync } from "node:fs";
import { dirname } from "node:path";
import { manifestPath } from "./layout.js";
import { extractH1, parseKvBullets, composeKvBullets, parseListSection, composeListSection } from "./mdMeta.js";
import type { WriteResult, ValidationIssue } from "./validate.js";

export interface ManifestFields {
  id?: string; version?: string; name?: string; description?: string;
  flows?: string[]; clock?: string; entry?: string;
}

function parseManifest(text: string): ManifestFields {
  const kv = parseKvBullets(text);
  const flows = parseListSection(text, "flows");
  const lines = text.split("\n");
  const h1Idx = lines.findIndex((l) => /^#\s+/.test(l));
  const desc: string[] = [];
  for (let i = h1Idx + 1; i < lines.length && h1Idx >= 0; i++) {
    const l = lines[i];
    if (/^-\s+/.test(l) || /^##\s+/.test(l)) break;
    if (l.trim()) desc.push(l.trim());
  }
  return {
    name: extractH1(text) ?? undefined,
    description: desc.join("\n") || undefined,
    id: kv.id, version: kv.version, clock: kv.clock, entry: kv.entry,
    flows: flows.length ? flows : undefined,
  };
}

function composeManifest(f: ManifestFields): string {
  const parts: string[] = [`# ${f.name ?? "(未命名)"}`];
  if (f.description) parts.push(f.description);
  const kv = composeKvBullets({ id: f.id, version: f.version, clock: f.clock, entry: f.entry });
  if (kv) parts.push(kv);
  const flows = composeListSection("flows", f.flows ?? []);
  if (flows) parts.push(flows);
  return parts.join("\n\n") + "\n";
}

export function setManifest(packDir: string, m: ManifestFields): WriteResult {
  const file = manifestPath(packDir);
  const existing = existsSync(file) ? parseManifest(readFileSync(file, "utf8")) : {};
  const merged: ManifestFields = { ...existing, ...m };
  mkdirSync(dirname(file), { recursive: true });
  writeFileSync(file, composeManifest(merged), "utf8");
  const warnings: ValidationIssue[] = [];
  if (!merged.id) warnings.push({ level: "warn", file: "manifest.md", msg: "缺 id(建库写 session_meta 用)", hint: "set_manifest 传 id" });
  if (!merged.flows?.length) warnings.push({ level: "warn", file: "manifest.md", msg: "未声明 flows" });
  return { file, warnings };
}
```

```ts
import { readFileSync, writeFileSync, existsSync, mkdirSync } from "node:fs";

import { serializeCsv, parseCsv } from "./csv.js";
import { checkCsvColumns } from "./validate.js";

const META_COLS = ["weight", "source", "visible"];

export function addPool(
  packDir: string,
  p: { pool: string; rows: Record<string, string | number>[] },
```

```ts
import { readFileSync, writeFileSync, existsSync, mkdirSync } from "node:fs";
import { dirname } from "node:path";
import { sheetPath } from "./layout.js";
import { serializeCsv, parseCsv } from "./csv.js";
import { checkCsvColumns } from "./validate.js";
import type { WriteResult } from "./validate.js";

const COLS = ["entity", "attr", "value", "visible"];

export function setSheetCell(
  packDir: string,
  s: { file?: string; cells: { entity: string; attr: string; value: string; visible?: 0 | 1 | 2 }[] },
): WriteResult {
  const file = sheetPath(packDir, s.file ?? "开局.csv");
  const existing = existsSync(file) ? parseCsv(readFileSync(file, "utf8")) : [];
  const byKey = new Map<string, Record<string, string | number>>();
  for (const r of existing) byKey.set(`${r.entity} ${r.attr}`, r);
  for (const c of s.cells) {
    byKey.set(`${c.entity} ${c.attr}`, {
      entity: c.entity, attr: c.attr, value: c.value, visible: c.visible ?? 0,
    });

  writeFileSync(file, serializeCsv(rows, COLS), "utf8");
  return { file, warnings: checkCsvColumns(`sheets/${s.file ?? "开局.csv"}`, rows, ["entity", "attr", "value"]) };
```

```ts
//

import { writeFileSync, mkdirSync } from "node:fs";
```

```ts
import { frontPath } from "./layout.js";
import { composeKvBullets, composeMdTable } from "./mdMeta.js";

export function addFront(
  packDir: string,
  f: {
```

```ts
// Copyright (C) 2026 MuLeiSY2021
//
// This file is part of Dicelore.
//
// Dicelore is free software: you can redistribute it and/or modify it under
// the terms of the GNU Affero General Public License as published by the Free
// Software Foundation, either version 3 of the License, or (at your option)
// any later version. See <https://www.gnu.org/licenses/>.

import { readFileSync, existsSync } from "node:fs";
import {
  manifestPath, worldDocPath, npcPath, poolPath, paramPath, rulePath, sheetPath, frontPath,
} from "./layout.js";
import { parseKvBullets } from "./mdMeta.js";

export type PackKind = "manifest" | "world" | "npc" | "pool" | "param" | "sheet" | "rule" | "front";

const MD_KINDS: PackKind[] = ["manifest", "world", "npc", "rule", "front"];

function pathFor(packDir: string, kind: PackKind, id?: string): string {
  switch (kind) {
    case "manifest": return manifestPath(packDir);
    case "world": return worldDocPath(packDir, id ?? "");
    case "npc": return npcPath(packDir, id ?? "");
    case "pool": return poolPath(packDir, id ?? "");
    case "param": return paramPath(packDir, id ?? "");
    case "rule": return rulePath(packDir, id ?? "");
    case "front": return frontPath(packDir, id ?? "");
    case "sheet": return sheetPath(packDir, id ?? "开局.csv");
  }
}

export function readPack(
  packDir: string,
  ref: { kind: PackKind; id?: string },
): { found: boolean; content: string; meta?: Record<string, string> } {
  const file = pathFor(packDir, ref.kind, ref.id);
  if (!existsSync(file)) return { found: false, content: "" };
  const content = readFileSync(file, "utf8");
  if (MD_KINDS.includes(ref.kind)) return { found: true, content, meta: parseKvBullets(content) };
  return { found: true, content };
}
```

```ts
// Copyright (C) 2026 MuLeiSY2021
//
// This file is part of Dicelore.
//
// Dicelore is free software: you can redistribute it and/or modify it under
// the terms of the GNU Affero General Public License as published by the Free
// Software Foundation, either version 3 of the License, or (at your option)
// any later version. See <https://www.gnu.org/licenses/>.

import { existsSync, readFileSync, readdirSync } from "node:fs";
import { join } from "node:path";
import { manifestPath, PACK_DIRS } from "./layout.js";
import { parseCsv } from "./csv.js";
import { parseKvBullets, parseListSection } from "./mdMeta.js";
import { checkCsvColumns, checkRangeBands, type ValidationIssue } from "./validate.js";

// 框架已发布的流程 skill 集合(对齐 packages/core/skills/dicelore-flow-*)。
export const KNOWN_FLOWS = [
  "dicelore-flow-gacha", "dicelore-flow-contest", "dicelore-flow-anka", "dicelore-flow-explore",
];

function csvFilesIn(dir: string): string[] {
  return existsSync(dir) ? readdirSync(dir).filter((x) => x.endsWith(".csv")) : [];
}

export function validatePack(packDir: string): { issues: ValidationIssue[]; ok: boolean } {
  const issues: ValidationIssue[] = [];

  const mf = manifestPath(packDir);
  if (!existsSync(mf)) {
    issues.push({ level: "error", file: "manifest.md", msg: "缺 manifest.md", hint: "set_manifest 至少写 id/name" });
  } else {
    const text = readFileSync(mf, "utf8");
    if (!parseKvBullets(text).id) issues.push({ level: "error", file: "manifest.md", msg: "manifest 缺 id" });
    for (const f of parseListSection(text, "flows")) {
      if (!KNOWN_FLOWS.includes(f)) {
        issues.push({ level: "error", file: "manifest.md", msg: `flows 引用了不存在的流程 skill「${f}」(FLOW_NOT_FOUND)`, hint: `可选: ${KNOWN_FLOWS.join(", ")}` });
      }
    }
  }

  for (const f of csvFilesIn(join(packDir, PACK_DIRS.params))) {
    const rows = parseCsv(readFileSync(join(packDir, PACK_DIRS.params, f), "utf8"));
    issues.push(...checkCsvColumns(`params/${f}`, rows, ["label", "min", "max", "consequence"]));
    issues.push(...checkRangeBands(`params/${f}`, rows.map((r) => ({ label: r.label, min: Number(r.min), max: Number(r.max) }))));
  }
  for (const f of csvFilesIn(join(packDir, PACK_DIRS.pools))) {
    issues.push(...checkCsvColumns(`pools/${f}`, parseCsv(readFileSync(join(packDir, PACK_DIRS.pools, f), "utf8")), []));
  }
  for (const f of csvFilesIn(join(packDir, PACK_DIRS.sheets))) {
    issues.push(...checkCsvColumns(`sheets/${f}`, parseCsv(readFileSync(join(packDir, PACK_DIRS.sheets, f), "utf8")), ["entity", "attr", "value"]));
  }

  return { issues, ok: !issues.some((i) => i.level === "error") };
}
```

```ts
// Copyright (C) 2026 MuLeiSY2021
//
// This file is part of Dicelore.
//
// Dicelore is free software: you can redistribute it and/or modify it under
// the terms of the GNU Affero General Public License as published by the Free
// Software Foundation, either version 3 of the License, or (at your option)
// any later version. See <https://www.gnu.org/licenses/>.

export interface Chunk {
  source: string;
  ordinal: number;
  content: string;
}

// 按空行分段,贪心聚合到 ≤maxChars(单段超长则自成一块)。
export function chunkText(text: string, source: string, opts: { maxChars?: number } = {}): Chunk[] {
  const max = opts.maxChars ?? 800;
  const paras = text.split(/\n\s*\n/).map((p) => p.trim()).filter(Boolean);
  const chunks: Chunk[] = [];
  let buf = "";
  let ordinal = 0;
  const flush = () => { if (buf) { chunks.push({ source, ordinal: ordinal++, content: buf }); buf = ""; } };
  for (const p of paras) {
    if (buf && buf.length + p.length + 2 > max) flush();
    buf = buf ? `${buf}\n\n${p}` : p;
  }
  flush();
  return chunks;
}
```

```ts
import Database from "better-sqlite3";

export type RetrievalDB = Database.Database;

// 临时检索库:成品包不含此库(构建期工件,可弃)。复用 store/fts 的 FTS5/jieba 基建。
export function openRetrievalDb(path: string): RetrievalDB {

  db.exec(`CREATE TABLE IF NOT EXISTS chunk (

  );`);
  db.exec(ftsTableDDL("chunk_fts"));
  return db;
}
```

```ts
//
// This file is part of Dicelore.
//
// Dicelore is free software: you can redistribute it and/or modify it under
// the terms of the GNU Affero General Public License as published by the Free
// Software Foundation, either version 3 of the License, or (at your option)
// any later version. See <https://www.gnu.org/licenses/>.

import { readFileSync } from "node:fs";
import { basename } from "node:path";
import { ftsIndex } from "../../store/fts.js";
import { chunkText, type Chunk } from "./chunk.js";
import type { RetrievalDB } from "./db.js";

export function ingest(db: RetrievalDB, src: { files?: string[]; text?: string }): { chunks: number } {
  const all: Chunk[] = [];
  if (src.text) all.push(...chunkText(src.text, "(inline)"));
  for (const f of src.files ?? []) all.push(...chunkText(readFileSync(f, "utf8"), basename(f)));
  const insert = db.prepare("INSERT INTO chunk (source, ordinal, content) VALUES (?, ?, ?)");
  const tx = db.transaction((chunks: Chunk[]) => {
    for (const c of chunks) {
      const info = insert.run(c.source, c.ordinal, c.content);
      ftsIndex(db, "chunk_fts", Number(info.lastInsertRowid), c.content);
    }
  });
  tx(all);
  return { chunks: all.length };
}
```

```ts
// Copyright (C) 2026 MuLeiSY2021
//
// This file is part of Dicelore.
//
// Dicelore is free software: you can redistribute it and/or modify it under
// the terms of the GNU Affero General Public License as published by the Free
// Software Foundation, either version 3 of the License, or (at your option)
// any later version. See <https://www.gnu.org/licenses/>.

import { ftsSearch } from "../../store/fts.js";
import type { RetrievalDB } from "./db.js";

export function search(
  db: RetrievalDB,
  q: { query: string; k?: number },
): { chunks: { source: string; content: string }[] } {
  const hits = ftsSearch(db, "chunk_fts", q.query, q.k ?? 8);
  const stmt = db.prepare("SELECT source, content FROM chunk WHERE rowid = ?");
  const chunks = hits
    .map((h) => stmt.get(h.rowid) as { source: string; content: string } | undefined)
    .filter((c): c is { source: string; content: string } => c !== undefined);
  return { chunks };
}
```
